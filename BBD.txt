const express = require('express');
const app = express();
const port = 3002;

// Define a GET endpoint
app.get('/api/data', (req, res) => {
  const data = {
    message: 'Hello, this is your data!',
    timestamp: new Date(),
  };
  res.json(data);
});

// Start the server
app.listen(port, () => {
  console.log(`Server is running on http://localhost:${port}`);
});
